  \begin{enumerate}

    \item EG01-24 Linearly Increasing Power Demand: 6000 MW
    \item EG01-29 Constant Power Demand: 0 MW
    \item EG01-30 Linearly Increasing Power Demand: 8000 MW 
\end{enumerate}
